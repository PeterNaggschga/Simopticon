14  operationsLock.lock();
15  if (upperValues.empty()) {
16  auto entry = tba.begin();

20 
21  for (const auto &entry: tba) {

30  long diff = (long) upperValues.size() - (long) lowerValues.size();
31  if (diff > 0) {
32  auto it = upperValues.begin();
33  advance(it, diff / 2);
34  lowerValues.insert(upperValues.begin(), it);
35  upperValues.erase(upperValues.begin(), it);
36  } else if (diff < 0) {
37  auto it = lowerValues.end();
38  advance(it, floor((float) diff / 2));
39  upperValues.insert(it, lowerValues.end());
40  lowerValues.erase(it, lowerValues.end());
41  }
