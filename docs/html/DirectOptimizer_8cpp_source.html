18 void DirectOptimizer::runOptimization() {
19  stopCon.setStartNow();
20  shared_ptr<HyRect> base(new BaseRect(D));
21  addActiveRects({base});
22  size_t evaluations = 2;
23  size_t numberOfRectangles = 1;
24  functionValue bestValue = getValueMap().getTopVals().front().second;
25  if (trackProgress) {
26  saveProgress(bestValue, evaluations, numberOfRectangles);
27  }
28 
29  while (!aborted && stopCon.evaluate(evaluations, numberOfRectangles, bestValue)) {
30  list<shared_ptr<HyRect>> newRects, oldRects;
31  for (const shared_ptr<HyRect> &rect: optimalRectangles(numberOfRectangles, bestValue)) {
32  for (const shared_ptr<HyRect> &newRect: HyRect::divide(rect)) {
33  newRects.push_back(newRect);
34  }
35  oldRects.push_back(rect);
36  numberOfRectangles += 2;
37  }
38  addActiveRects(newRects);
39  removeActiveRects(oldRects);
40  static functionValue oldPhi = bestValue;
41  static unsigned char phaseIters = 0;
42  bestValue = getValueMap().getTopVals().front().second;
43  phaseIters++;
44  level.nextLevel();
45  if (level.isGlobal()) {
46  if (bestValue < oldPhi - Levels::L3_EPSILON * abs(oldPhi - getValueMap().getMedian()) || phaseIters > 35) {
47  level.setGlobal(false);
48  oldPhi = bestValue;
49  phaseIters = 0;
50  }
51  }
52  if (!level.isGlobal() && phaseIters > 3) {
53  level.setGlobal(bestValue > oldPhi - Levels::L3_EPSILON * abs(oldPhi - getValueMap().getMedian()));
54  oldPhi = bestValue;
55  phaseIters = 0;
56  }
57  evaluations = getValueMap().getSize();
58  iterations++;
59  if (trackProgress) {
60  saveProgress(bestValue, evaluations, numberOfRectangles);
61  }
62  }
63  if (printValues) {
64  saveValues();
65  }
66 }

106 void DirectOptimizer::addActiveRects(const list<shared_ptr<HyRect>> &rects) {
107  struct hashFunction {
108  size_t operator()(const vector<dirCoordinate> &v) const {
109  hash<dirCoordinate> hash;
110  size_t res = 0;
111  for (dirCoordinate cord: v) {
112  res ^= hash(cord);
113  }
114  return hash(res);
115  }
116  };
117 
118  map<shared_ptr<HyRect>, array<vector<dirCoordinate>, 2>> rectToVec;
119  unordered_set<vector<dirCoordinate>, hashFunction> vecs;
120  for (const auto &rect: rects) {
121  auto vertices = rect->getSamplingVertices();
122  rectToVec.insert(make_pair(rect, vertices));
123  vecs.insert(vertices.begin(), vertices.end());
124  }
125 
126  auto samples = getValues(list<vector<dirCoordinate>>(vecs.begin(), vecs.end()));
127 
128  for (const auto &entry: rectToVec) {
129  entry.first->setAvgValue((samples[entry.second[0]] + samples[entry.second[1]]) / 2);
130  depth depth = entry.first->getDepth();
131  auto it = activeRects.find(depth);
132  if (it != activeRects.end()) {
133  it->second.insert(entry.first);
134  } else {
135  set<shared_ptr<HyRect>, CmpSharedHyrect> newSet = {entry.first};
136  activeRects.insert(make_pair(depth, newSet));
137  }
138  }
139 }
