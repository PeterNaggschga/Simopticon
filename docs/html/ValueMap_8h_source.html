18 /**
19  * A container managing a map data structure that maps Parameter combinations to their respective found values.
20  * The class manages concurrent access using the #operationsLock.
21  * Running median calculation is supported by using sets #upperValues and #lowerValues.
22  * Values are inserted into the data structure at once when #updateMap is called.
23  * Before that they are saved in #tba to avoid unnecessary costly insertion operations.
24  * @ingroup controller
25  */
26 class ValueMap {
27 private:
28  /**
29  * Threadlock to avoid damage to the data structure when concurrent threads access it.
30  */
31  mutex operationsLock;
32 
33  /**
34  * Greater half of the values in #values. Same size as or one element more than #lowerValues.
35  */
36  set<functionValue *, CmpPtrFunctionvalue> upperValues;
37  /**
38  * Lesser half of the values in #values. Same size as or one element less than #upperValues.
39  */
40  set<functionValue *, CmpPtrFunctionvalue> lowerValues;
41 
42  /**
43  * Number of entries to be printed as best values at the end of the optimization process.
44  * Can be configured in main config.
45  */
46  const unsigned int topEntries;
47  /**
48  * Set of pairs of the best Parameter combinations and their respective values.
49  * Contains not more than #topEntries entries.
50  */
51  set<pair<const vector<shared_ptr<Parameter>>, functionValue>, CmpPairVectorSharedParameterFunctionvalue> topVals;
52 
53  /**
54  * Actual map that contains Parameter combinations and their respective values.
55  */
56  map<vector<shared_ptr<Parameter>>, functionValue, CmpVectorSharedParameter> values;
57 
58  /**
59  * Entries that have been added since last #updateMap.
60  * Will be inserted into #values, #upperValues and #lowerValues when #updateMap is called.
61  */
62  list<pair<vector<shared_ptr<Parameter>>, functionValue>> tba;
63 
64  /**
65  * Takes all values in #tba, adds them to #lowerValues or #upperValues and inserts them into #values.
66  * #lowerValues and #upperValues are sorted as is required by their constraints.
67  * Afterwards #tba is cleared.
68  */
69  void updateMap();
70 
71  /**
72  * Inserts a single value into #values and into #lowerValues or #upperValues depending on @a set argument.
73  * @param val: Parameter combination and respective value to be inserted.
74  * @param set: Set that value is inserted in. Either #lowerValues or #upperValues.
75  */
76  void addValue(const pair<vector<shared_ptr<Parameter>>, functionValue> &val,
77  set<functionValue *, CmpPtrFunctionvalue> &set);
78 
79 public:
80  /**
81  * Creates a ValueMap.
82  * @param topEntries: Value to be assigned to #topEntries.
83  */
84  explicit ValueMap(unsigned int topEntries = 10);
85 
86  /**
87  * Returns the value saved at the given Parameter combination. If no value is present, an exception is thrown.
88  * Triggers #updateMap.
89  * @param params: Parameter combination to which the value is requested.
90  * @return The value saved in #values at the given Parameter combination.
91  */
92  [[nodiscard]] functionValue query(const vector<shared_ptr<Parameter>> &params);
93 
94  /**
95  * Adds the given Parameter combination and value to #tba.
96  * @param params: Parameter combination to be added.
97  * @param val: Value to be added.
98  */
99  void insert(const vector<shared_ptr<Parameter>> &params, functionValue val);
100 
101  /**
102  * Checks if a value has been recorded at the given Parameter combination.
103  * Triggers #updateMap.
104  * @param cords: Parameter combination that is checked.
105  * @return A boolean value that represents if the value is known.
106  */
107  [[nodiscard]] bool isKnown(const vector<shared_ptr<Parameter>> &cords);
108 
109  /**
110  * Checks if the given Parameter combination is to be found in #topVals.
111  * Triggers #updateMap.
112  * @param cords: Parameter combination that is checked.
113  * @return A boolean value that represents if the value is one of the best #topEntries entries in #values.
114  */
115  [[nodiscard]] bool isTopValue(const vector<shared_ptr<Parameter>> &cords);
116 
117  /**
118  * Returns the whole #values member.
119  * Triggers #updateMap.
120  * @return A map reference to #values.
121  */
122  const map<vector<shared_ptr<Parameter>>, functionValue, CmpVectorSharedParameter> &getValues();
123 
124  /**
125  * Returns the median of all values using #lowerValues and #upperValues.
126  * If no values have been added, 0 is returned.
127  * Triggers #updateMap.
128  * @return A value representing the median of all values.
129  */
130  [[nodiscard]] functionValue getMedian();
131 
132  /**
133  * Returns the number of inserted values. Values in #tba are included.
134  * @return An integral representing the number of inserted values.
135  */
136  [[nodiscard]] size_t getSize() const;
137 
138  /**
139  * Returns the best #topEntries entries that are saved in #topVals.
140  * Triggers #updateMap.
141  * @return A list of the best #topEntries Parameter combinations and their respective values.
142  */
143  [[nodiscard]] list<pair<vector<shared_ptr<Parameter>>, functionValue>> getTopVals();
144 };
