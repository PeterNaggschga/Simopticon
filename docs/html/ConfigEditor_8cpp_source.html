47 void ConfigEditor::replaceOption(std::string &file, std::string option, const std::string &value) {
48  option = "\n" + option + " = ";
49  size_t pos = file.find(option);
50  if (pos == std::string::npos) {
51  const std::string general = "[General]\n";
52  pos = file.find(general);
53  file.insert(pos + general.size(), option + value + "\n");
54  return;
55  }
56  pos += option.size();
57  size_t endOfLine = file.find('\n', pos);
58  file.replace(pos, endOfLine - pos, value);
