128 std::map<parameterCombination, functionValue> Controller::requestValues(const std::list<parameterCombination> &params) {
129  updateStatus();
130  std::map<parameterCombination, functionValue> result;
131  std::set<parameterCombination, CmpVectorSharedParameter> simRuns;
132  for (const auto &cords: params) {
133  if (valueMap->isKnown(cords)) {
134  result.insert(make_pair(cords, valueMap->query(cords)));
135  } else {
136  simRuns.insert(cords);
137  }
138  }
139 
140  if (!simRuns.empty()) {
141  stepState.next();
142  auto vecToResult = runSimulations(simRuns);
143  stepState.next();
144  auto simResults = evaluate(vecToResult);
145  stepState.next();
146 
147  for (const auto &entry: simResults) {
148  if (!aborted) {
149  valueMap->insert(entry.first, entry.second);
150  }
151  topResults.insert(make_pair(entry.first, vecToResult[entry.first].first));
152  }
153 
154  removeOldResultfiles();
155 
156  result.merge(simResults);
157  }
158  updateStatus();
159  return result;
160 }

166 void Controller::run() {
167  auto runStatusUpdate = [this]() {
168  while (statusInterval != std::chrono::milliseconds(0)) {
169  updateStatus();
170  std::this_thread::sleep_for(statusInterval);
171  }
172  };
173  stepState.next();
174  updateStatus();
175  std::thread statusThread(runStatusUpdate);
176  optimizer->runOptimization();
177  statusInterval = std::chrono::milliseconds(0);
178  statusThread.join();
179  std::list<std::pair<parameterCombination, std::pair<functionValue, std::filesystem::path>>> top;
180  for (const auto &entry: valueMap->getTopVals()) {
181  top.emplace_back(entry.first, std::make_pair(entry.second, topResults[entry.first]));
182  }
183  StatusBar::printResults(top);
184 }

194 std::map<parameterCombination, functionValue, CmpVectorSharedParameter> Controller::evaluate(

196  updateStatus();
197  std::set<std::pair<std::filesystem::path, std::set<runId>>> resultFiles;

199  resultFiles.insert(entry.second);

202  std::map<parameterCombination, functionValue, CmpVectorSharedParameter> result;
203  for (const auto &entry: simulationResults) {
204  result.insert(make_pair(entry.first, evaluationResult[entry.second]));
205  }
206  updateStatus();

212  std::filesystem::path resultDir = topResults.begin()->second.parent_path();
213  std::filesystem::remove_all(resultDir);
214  } else if (topResults.size() > valueMap->getTopVals().size()) {

216  for (const auto &entry: topResults) {

218  std::filesystem::remove_all(entry.second);
219  toBeRemoved.push_back(entry.first);
220  }

224  }
