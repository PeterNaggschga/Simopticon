119  updateStatus();
120  map<vector<shared_ptr<Parameter>>, functionValue> result;
121  set<vector<shared_ptr<Parameter>>, CmpVectorSharedParameter> simRuns;
122  for (const auto &cords: params) {
123  if (valueMap->isKnown(cords)) {
124  result.insert(make_pair(cords, valueMap->query(cords)));
125  } else {
126  simRuns.insert(cords);

128  }
129 
130  if (!simRuns.empty()) {
131  stepState.next();
132  auto vecToResult = runSimulations(simRuns);
133  stepState.next();

135  stepState.next();
136 
137  for (const auto &entry: simResults) {
138  if (!aborted) {
139  valueMap->insert(entry.first, entry.second);
140  }
141  topResults.insert(make_pair(entry.first, vecToResult[entry.first].first));
142  }
143 
144  removeOldResultfiles();
145 
146  result.merge(simResults);
147  }

156 void Controller::run() {
157  auto runStatusUpdate = [this]() {
158  while (statusInterval != milliseconds(0)) {
159  updateStatus();
160  this_thread::sleep_for(statusInterval);
161  }
162  };
163  stepState.next();
164  updateStatus();
165  thread statusThread(runStatusUpdate);
166  optimizer->runOptimization();
167  statusInterval = milliseconds(0);
168  statusThread.join();
169  list<pair<vector<shared_ptr<Parameter>>, pair<functionValue, filesystem::path>>> top;
170  for (const auto &entry: valueMap->getTopVals()) {
171  top.emplace_back(entry.first, make_pair(entry.second, topResults[entry.first]));
172  }
173  StatusBar::printResults(top);
174 }

184 map<vector<shared_ptr<Parameter>>, functionValue, CmpVectorSharedParameter> Controller::evaluate(
185  const map<vector<shared_ptr<Parameter>>, pair<filesystem::path, set<runId>>, CmpVectorSharedParameter> &simulationResults) {
186  updateStatus();
187  set<pair<filesystem::path, set<runId>>> resultFiles;
188  for (const auto &entry: simulationResults) {
189  resultFiles.insert(entry.second);
190  }
191  auto evaluationResult = evaluation->processOutput(resultFiles);
192  map<vector<shared_ptr<Parameter>>, functionValue, CmpVectorSharedParameter> result;
193  for (const auto &entry: simulationResults) {
194  result.insert(make_pair(entry.first, evaluationResult[entry.second]));
195  }
196  updateStatus();
197  return result;

200 void Controller::removeOldResultfiles() {
201  if (!(keepFiles || topResults.empty())) {
202  filesystem::path resultDir = topResults.begin()->second.parent_path();
203  filesystem::remove_all(resultDir);
204  } else if (topResults.size() > valueMap->getTopVals().size()) {
205  list<vector<shared_ptr<Parameter>>> toBeRemoved;
206  for (const auto &entry: topResults) {
207  if (!valueMap->isTopValue(entry.first)) {
208  filesystem::remove_all(entry.second);
209  toBeRemoved.push_back(entry.first);
210  }
211  }
212  for (const auto &entry: toBeRemoved) {
213  topResults.erase(entry);
214  }
215  }
216 }
