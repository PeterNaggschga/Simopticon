39 bool StoppingCondition::evaluate(size_t evaluations, size_t hyrects, functionValue newBestVal) {
40  bool eval = NR_EVALUATIONS == 0 || evaluations < NR_EVALUATIONS;
41  bool rects = NR_HYRECTS == 0 || hyrects < NR_HYRECTS;
42  bool time = !time_eval || END_TIME > std::chrono::system_clock::now();
43  bool accuracy = updateAccuracy(newBestVal) || (ACCURACY == 0 && NR_ACCURACY_ITERATIONS == 0);
